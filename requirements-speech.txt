# Phase 3 speech-pipeline extras. Added to the image when the Docker build
# arg SPEECH_STACK=full (the default). To skip — smaller image, faster-whisper
# fallback path only, no VAD-based chunking or forced alignment:
#
#     docker compose build --build-arg SPEECH_STACK=slim
#
# whisperx pulls in faster-whisper (already an image dep), pyannote-audio,
# torchaudio, and the wav2vec2 alignment weights. ~500 MB added.
#
# demucs is optional vocal-stem separation (scripts/lib/vocal_sep.py). Only
# loaded at runtime when config/speech.json::vocal_separation.enabled = true.
# ~80 MB for the package; model weights (~1 GB htdemucs_ft) download on
# first use.

whisperx>=3.1
demucs>=4.0

# torchcodec is pyannote.audio 4.x's preferred audio decoder. It has NO torch
# version pin but is ABI-coupled — 0.7.0 matches torch 2.8 (the venv's version).
# It also needs FFmpeg *shared* libraries at runtime; the project's static
# C:\ffmpeg\bin has none, so scripts/lib/speech.py discovers a shared set
# (e.g. the AMD GPU driver's CNext dir) and puts it on the DLL search path
# before import. If unavailable, pyannote falls back to soundfile (Stage 2
# pre-extracts WAV, so the fallback is fine). See
# AIclippingPipelineVault/wiki/concepts/clip-quality-remediation-2026-06.md Fix 4.
torchcodec==0.7.0
